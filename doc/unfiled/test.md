class Solution {
    /**
    * @param {int[]} nums
    * @param {int} target
    * @return {int[]}
    */
    public int[] twoSum(int[] nums, int target) {
        int len = nums.length;
        for (int i=0; i&lt;len; i++) {
            for (int j=i+1; j&lt;len; j++) {
                if (nums[i]+nums[j]==target)
                    return new int[]{i, j};
            }
        }
        return null;
    }
}
